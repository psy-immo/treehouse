     * add the bottom element of the concept lattice (a concept lattice is

    last = addConceptToBulk(root, M);

    nextClosure:

    for (int i = c->attributes - 1; i >= 0; --i)
    {

        if (!INCIDES(M[i]))

            closeIntent(ctx, M, Y);

            int good;

            for (int j = 0; j < i; ++j)
            {
                if (INCIDES(Y[j]))
                {
                    if (!INCIDES((M[j])))
                    {

                IncidenceCell *DELTA;

                Y = DELTA;

                goto nextClosure;
