    /*
     * calculate the bottom intent of the concept lattice, i.e. {}''

    closeIntent(ctx, Y, M);

    int count;

    count = 1;

    /*
     * begin of nextClosure function iteration
     */
    nextClosure:

    for (int i = c->attributes - 1; i >= 0; --i)
    {

        if (!INCIDES(M[i]))
        {
            CROSS(M[i]);
            closeIntent(ctx, M, Y);

            int good;
            good = 1;

            for (int j = 0; j < i; ++j)
            {
                if (INCIDES(Y[j]))
                {
                    if (!INCIDES((M[j])))
                    {

                        good = 0;
                        break;
                    }
                }
            }
            if (good)
            {
                /*
                 * we found the next intent
                 */
                count++;

                /*
                 * continue with Y for M
                 */

                IncidenceCell *DELTA;
                DELTA = M;
                M = Y;
                Y = DELTA;
                /*
                 * do the nextClosure
                 */
                goto nextClosure;
            }
        }

        CLEAR(M[i]);
    }

    /*
     * free up memory
     */

    free(M);
    free(Y);
